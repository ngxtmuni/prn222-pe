1. Q1 – Console Client hoàn chỉnh
File: Program.cs
using System.Net.Sockets;
using System.Text;
using System.Text.Json;

namespace Q1_BookTrackingClient
{
    internal class Program
    {
        private const string Host = "127.0.0.1";
        private const int Port = 4000;

        private static readonly JsonSerializerOptions JsonOptions = new()
        {
            PropertyNameCaseInsensitive = true
        };

        static async Task Main(string[] args)
        {
            while (true)
            {
                Console.Write("Enter Book ID (or press Enter to exit): ");
                string? input = Console.ReadLine();

                if (string.IsNullOrWhiteSpace(input))
                {
                    Console.WriteLine("Goodbye! Book tracking client is shutting down.");
                    break;
                }

                if (!int.TryParse(input, out int bookId) || bookId <= 0)
                {
                    Console.WriteLine("Invalid input! Please enter a valid Book ID (positive integer).");
                    continue;
                }

                try
                {
                    string jsonResponse = await SendBookIdToServerAsync(bookId);
                    ProcessServerResponse(jsonResponse, bookId);
                }
                catch
                {
                    Console.WriteLine("Book tracking server is not running. Please try again later.");
                }
            }
        }

        private static async Task<string> SendBookIdToServerAsync(int bookId)
        {
            using TcpClient client = new();
            await client.ConnectAsync(Host, Port);

            using NetworkStream stream = client.GetStream();
            using StreamWriter writer = new(stream, Encoding.UTF8, leaveOpen: true);
            using StreamReader reader = new(stream, Encoding.UTF8, leaveOpen: true);

            writer.AutoFlush = true;

            await writer.WriteLineAsync(bookId.ToString());

            string response = await reader.ReadToEndAsync();
            return response.Trim();
        }

        private static void ProcessServerResponse(string json, int bookId)
        {
            if (string.IsNullOrWhiteSpace(json))
            {
                Console.WriteLine("No response from server.");
                return;
            }

            try
            {
                using JsonDocument doc = JsonDocument.Parse(json);
                JsonElement root = doc.RootElement;

                // Case 1: server trả []
                if (root.ValueKind == JsonValueKind.Array)
                {
                    List<BorrowerInfo>? borrowers =
                        JsonSerializer.Deserialize<List<BorrowerInfo>>(json, JsonOptions);

                    if (borrowers == null || borrowers.Count == 0)
                    {
                        Console.WriteLine($"No borrower records found for Book ID {bookId}.");
                        return;
                    }

                    Console.WriteLine($"=== Borrower History for Book ID: {bookId}");

                    foreach (var item in borrowers)
                    {
                        Console.WriteLine($"Reader ID: {item.ReaderId}");
                        Console.WriteLine($"Full Name: {item.FullName}");
                        Console.WriteLine($"Email: {item.Email}");
                        Console.WriteLine($"Borrow Date: {item.BorrowDate:yyyy-MM-dd}");
                        Console.WriteLine($"Return Date: {(item.ReturnDate.HasValue ? item.ReturnDate.Value.ToString("yyyy-MM-dd") : "Not returned yet")}");
                        Console.WriteLine($"Status: {item.Status}");
                        Console.WriteLine("---");
                    }

                    return;
                }

                // Case 2: server trả object kiểu { "message": "Book not found" }
                if (root.ValueKind == JsonValueKind.Object)
                {
                    if (root.TryGetProperty("message", out JsonElement messageElement))
                    {
                        string? message = messageElement.GetString();

                        if (!string.IsNullOrWhiteSpace(message) &&
                            message.Contains("book not found", StringComparison.OrdinalIgnoreCase))
                        {
                            Console.WriteLine($"Book with ID {bookId} does not exist.");
                            return;
                        }

                        Console.WriteLine(message);
                        return;
                    }
                }

                // fallback nếu server trả text lạ
                if (json.Contains("book not found", StringComparison.OrdinalIgnoreCase))
                {
                    Console.WriteLine($"Book with ID {bookId} does not exist.");
                    return;
                }

                Console.WriteLine(json);
            }
            catch
            {
                if (json.Contains("book not found", StringComparison.OrdinalIgnoreCase))
                {
                    Console.WriteLine($"Book with ID {bookId} does not exist.");
                    return;
                }

                Console.WriteLine("Invalid response from server.");
            }
        }
    }

    public class BorrowerInfo
    {
        public int ReaderId { get; set; }
        public string FullName { get; set; } = "";
        public string Email { get; set; } = "";
        public DateTime BorrowDate { get; set; }
        public DateTime? ReturnDate { get; set; }
        public string Status { get; set; } = "";
    }
}

2.5 Pages/Employee/List.cshtml.cs
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.RazorPages;
using Microsoft.EntityFrameworkCore;
using YourProject.Data;
using YourProject.Models;

namespace YourProject.Pages.Employee
{
    public class ListModel : PageModel
    {
        private readonly YourDbContext _context;

        public ListModel(YourDbContext context)
        {
            _context = context;
        }

        public List<Department> Departments { get; set; } = new();
        public List<Models.Employee> Employees { get; set; } = new();
        public List<EmployeeSkill> EmployeeSkills { get; set; } = new();

        [BindProperty(SupportsGet = true)]
        public int DepartmentId { get; set; } = 0;

        [BindProperty(SupportsGet = true)]
        public int? EmployeeId { get; set; }

        public string SelectedEmployeeName { get; set; } = "";

        public async Task OnGetAsync()
        {
            Departments = await _context.Departments
                .OrderBy(d => d.DepartmentName)
                .ToListAsync();

            IQueryable<Models.Employee> employeeQuery = _context.Employees
                .Include(e => e.Department);

            if (DepartmentId > 0)
            {
                employeeQuery = employeeQuery.Where(e => e.DepartmentId == DepartmentId);
            }

            Employees = await employeeQuery
                .OrderBy(e => e.Name)
                .ToListAsync();

            if (EmployeeId.HasValue)
            {
                Models.Employee? employee = await _context.Employees
                    .FirstOrDefaultAsync(e => e.EmployeeId == EmployeeId.Value);

                if (employee != null)
                {
                    SelectedEmployeeName = employee.Name;

                    EmployeeSkills = await _context.EmployeeSkills
                        .Where(es => es.EmployeeId == EmployeeId.Value)
                        .Include(es => es.Skill)
                        .OrderBy(es => es.Skill!.SkillName)
                        .ToListAsync();
                }
            }
        }
    }
}

Nếu đề cho DB script và bạn muốn scaffold nhanh:

dotnet add package Microsoft.EntityFrameworkCore.SqlServer
dotnet add package Microsoft.EntityFrameworkCore.Tools
dotnet add package Microsoft.VisualStudio.Web.CodeGeneration.Design

Scaffold:

Scaffold-DbContext "Server=.;Database=YourDatabaseName;Trusted_Connection=True;TrustServerCertificate=True" Microsoft.EntityFrameworkCore.SqlServer -OutputDir Models -ContextDir Data -Context YourDbContext -DataAnnotations -Force

Hoặc CLI:

dotnet ef dbcontext scaffold "Server=.;Database=YourDatabaseName;Trusted_Connection=True;TrustServerCertificate=True" Microsoft.EntityFrameworkCore.SqlServer --output-dir Models --context-dir Data --context YourDbContext --data-annotations --force
